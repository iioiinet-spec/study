📐 백준 2562 · 최댓값

9개의 서로 다른 자연수 중 최댓값과 위치(번째)를 구하는 문제 | 6가지 C 구현 기법

📋 문제 설명

🎯 입력 : 9개의 자연수 (각 ≤ 100)
📤 출력 : 최댓값, 몇 번째 수인지
💡 예시
입력: 3, 29, 38, 12, 57, 74, 40, 85, 61 → 출력: 85 (8번째)
✅ 9개의 수를 읽고 최댓값과 위치를 찾는 간결한 로직. 아래 6가지 스타일로 구현했습니다.
✅ 기본 풀이 (배열 없이 입력 즉시 처리)
#include <stdio.h>

int main(void) {
    int x, max, pos;
    for (int i = 1; i <= 9; i++) {
        scanf("%d", &x);
        if (i == 1) { max = x; pos = 1; }
        else if (x > max) { max = x; pos = i; }
    }
    printf("%d\n%d\n", max, pos);
    return 0;
}

📍 포인터 연산 *(p+i)
#include <stdio.h>
void findMaxPos(int *p, int n, int *max, int *pos) {
    *max = *p; *pos = 1;
    for (int i = 1; i < n; i++)
        if (*(p + i) > *max) { *max = *(p + i); *pos = i + 1; }
}
int main() {
    int arr[9], max, pos;
    for (int i = 0; i < 9; i++) scanf("%d", &arr[i]);
    findMaxPos(arr, 9, &max, &pos);
    printf("%d\n%d\n", max, pos);
    return 0;
}

🏛️ 구조체로 묶어서 반환
#include <stdio.h>
typedef struct { int value; int pos; } Info;
Info findMaxPos(int arr[], int n) {
    Info ans = {arr[0], 1};
    for (int i = 1; i < n; i++)
        if (arr[i] > ans.value) { ans.value = arr[i]; ans.pos = i + 1; }
    return ans;
}
int main() {
    int arr[9];
    for (int i = 0; i < 9; i++) scanf("%d", &arr[i]);
    Info ans = findMaxPos(arr, 9);
    printf("%d\n%d\n", ans.value, ans.pos);
    return 0;
}

🧠 구조체 배열 + malloc / free
#include <stdio.h>
#include <stdlib.h>
typedef struct { int num; } Data;
typedef struct { int value; int pos; } Info;
void findMaxPos(Data *p, int n, Info *ans) {
    ans->value = p[0].num; ans->pos = 1;
    for (int i = 1; i < n; i++)
        if ((p + i)->num > ans->value) { ans->value = (p + i)->num; ans->pos = i + 1; }
}
int main() {
    Data *arr = (Data*)malloc(sizeof(Data) * 9);
    for (int i = 0; i < 9; i++) scanf("%d", &arr[i].num);
    Info ans;
    findMaxPos(arr, 9, &ans);
    printf("%d\n%d\n", ans.value, ans.pos);
    free(arr);
    return 0;
}
