정올 3281 불쾌한 날2

한 구간을 딱 한 번 뒤집어서, 소들이 볼 수 있는 머리 모양 수의 총합을 최소로 만드는 문제입니다.

1문제 핵심

소들은 모두 오른쪽, 즉 동쪽을 봅니다.
어떤 소는 자기보다 키가 작은 소들을 볼 수 있지만, 자기보다 크거나 같은 소가 나오면 그 뒤는 볼 수 없습니다.

이 문제에서는 임의의 한 구간 s ~ e를 한 번 뒤집습니다.
뒤집은 뒤 전체 소들이 볼 수 있는 머리 모양 수의 합이 최소가 되도록 해야 합니다.
주의 : 정올 1141 불쾌한 날은 현재 배열의 총합만 구하지만, 3281 불쾌한 날2는 구간을 뒤집은 뒤의 최솟값을 구합니다.

2직접 뒤집어 보기

수업용 실험은 이해를 위해 직접 뒤집어 계산합니다. 정답 코드는 더 빠른 O(N²) 아이디어를 사용합니다.

3실행 결과

버튼을 누르면 결과가 표시됩니다.

5각 소가 볼 수 있는 수

6왜 단순 풀이는 느릴까?

가장 쉬운 방법은 모든 구간을 뒤집어 보고, 매번 전체 불쾌한 정도를 다시 계산하는 것입니다.

모든 구간 선택 : O(N²)
각 구간마다 다시 계산 : O(N)
전체 시간 : O(N³)
N = 1000이면 약 10억 번 가까운 계산이 될 수 있어 1초 제한에서 위험합니다.

7빠른 풀이 핵심

정답 코드는 배열을 매번 실제로 뒤집지 않습니다.

최종값 = 원래 불쾌한 정도 + 뒤집었을 때의 변화량

원래 보이던 쌍이 사라지면 -1, 새로 보이는 쌍이 생기면 +1을 더합니다.

많은 구간에 같은 값을 빠르게 더하기 위해 2차원 차이 배열 diff를 사용합니다.

8코드 작성 단계

1단계
소의 키를 배열 h[]에 저장합니다.
2단계
원래 상태에서 보이는 쌍을 계산하고, 원래 불쾌한 정도 original을 구합니다.
3단계
구간을 뒤집었을 때 사라지는 관계는 -1, 새로 생기는 관계는 +1로 변화량을 저장합니다.
4단계
add_rect() 함수로 여러 구간에 같은 변화량을 한 번에 더합니다.
5단계
2차원 누적합으로 diff[s][e]를 완성합니다.
6단계
original + diff[s][e] 중 최솟값을 출력합니다.

9핵심 함수 add_rect

이 함수는 구간 s ~ e의 후보들을 직사각형 범위로 보고, 그 범위 전체에 변화량을 더합니다.

void add_rect(int r1, int r2, int c1, int c2, long long v) {
    if (r1 > r2 || c1 > c2) return;

    if (r1 < 0) r1 = 0;
    if (c1 < 0) c1 = 0;
    if (r2 >= N) r2 = N - 1;
    if (c2 >= N) c2 = N - 1;

    if (r1 > r2 || c1 > c2) return;

    diff[r1][c1] += v;
    diff[r2 + 1][c1] -= v;
    diff[r1][c2 + 1] -= v;
    diff[r2 + 1][c2 + 1] += v;
}

10정답 C 코드

#include <stdio.h>

#define MAXN 1005

int N;
long long h[MAXN];
long long diff[MAXN + 2][MAXN + 2];
int visible[MAXN][MAXN];

void add_rect(int r1, int r2, int c1, int c2, long long v) {
    if (r1 > r2 || c1 > c2) return;

    if (r1 < 0) r1 = 0;
    if (c1 < 0) c1 = 0;
    if (r2 >= N) r2 = N - 1;
    if (c2 >= N) c2 = N - 1;

    if (r1 > r2 || c1 > c2) return;

    diff[r1][c1] += v;
    diff[r2 + 1][c1] -= v;
    diff[r1][c2 + 1] -= v;
    diff[r2 + 1][c2 + 1] += v;
}

int main() {
    scanf("%d", &N);

    for (int i = 0; i < N; i++) {
        scanf("%lld", &h[i]);
    }

    long long original = 0;

    for (int i = 0; i < N; i++) {
        long long mx = -1;

        for (int j = i + 1; j < N; j++) {
            if (h[i] > h[j] && h[i] > mx) {
                visible[i][j] = 1;
                original++;
            }

            if (h[j] > mx) {
                mx = h[j];
            }
        }
    }

    for (int i = 0; i < N; i++) {
        for (int j = i + 1; j < N; j++) {
            if (visible[i][j]) {
                add_rect(0, i, i, N - 1, -1);
                add_rect(0, j, j, N - 1, -1);
                add_rect(0, i, j, N - 1, 1);
            }
        }
    }

    for (int right = 0; right < N; right++) {
        long long mx = -1;

        for (int left = right - 1; left >= 0; left--) {
            if (h[left] > mx) {
                mx = h[left];
            }

            if (h[right] > mx) {
                add_rect(0, left, right, N - 1, 1);
            }
        }
    }

    for (int a = 0; a < N; a++) {
        int firstBlock = N;

        for (int x = a + 1; x < N; x++) {
            if (h[x] >= h[a]) {
                firstBlock = x;
                break;
            }
        }

        int nextBlock = N;

        for (int i = N - 1; i > a; i--) {
            if (h[i] >= h[a]) {
                nextBlock = i;
            }

            int s1 = a + 1;
            int s2 = i;

            if (firstBlock < s2) {
                s2 = firstBlock;
            }

            int e1 = i;
            int e2 = nextBlock - 1;

            add_rect(s1, s2, e1, e2, 1);
        }
    }

    for (int i = 0; i < N; i++) {
        int prevBlock = -1;

        for (int x = i - 1; x >= 0; x--) {
            if (h[x] >= h[i]) {
                prevBlock = x;
                break;
            }
        }

        int lastBlock = -1;

        for (int b = i + 1; b < N; b++) {
            if (h[b] >= h[i]) {
                lastBlock = b;
            }

            int s1 = prevBlock + 1;
            int s2 = i;
            int e1 = i;

            if (lastBlock > e1) {
                e1 = lastBlock;
            }

            int e2 = b - 1;

            add_rect(s1, s2, e1, e2, 1);
        }
    }

    long long answer = 4000000000000000000LL;

    for (int s = 0; s < N; s++) {
        long long rowSum = 0;

        for (int e = 0; e < N; e++) {
            rowSum += diff[s][e];

            if (s > 0) {
                diff[s][e] = diff[s - 1][e] + rowSum;
            } else {
                diff[s][e] = rowSum;
            }

            if (s < e) {
                long long now = original + diff[s][e];

                if (now < answer) {
                    answer = now;
                }
            }
        }
    }

    printf("%lld\n", answer);

    return 0;
}
