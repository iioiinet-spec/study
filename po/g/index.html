골드 1. 두 다항식 곱셈

계수 배열로 저장된 두 다항식을 곱합니다. 핵심은 모든 항을 서로 곱하고, 결과 차수에 누적하는 것입니다.

C[i + j] += A[i] × B[j]

x^i × x^j = x^(i+j)이므로 결과 배열의 위치는 i+j가 됩니다.

1. 입력값 바꾸기

낮은 차수부터 입력합니다. 예: 1 2 3 → 3x^2 + 2x + 1

3. 출력 결과

4. 단계별 곱셈

골드 2. 희소 다항식 덧셈

차수가 크고 항이 적은 다항식은 배열 전체를 만들지 않고, 0이 아닌 항만 구조체 배열로 저장합니다.

Term = { coef, exp }

두 다항식의 항이 차수 내림차순으로 주어지므로, 두 포인터 i, j로 병합하면 됩니다.

1. 입력값 바꾸기

한 줄에 하나의 항을 입력합니다. 형식: 계수 차수

2. 항 배열 비교

3. 출력 결과

4. 단계별 병합

골드 1. 다항식 곱셈

#include <stdio.h>

void printPoly(int coef[], int degree)
{
    int first = 1;

    for(int i = degree; i >= 0; i--)
    {
        if(coef[i] == 0) continue;

        if(!first) printf(" + ");

        if(i == 0)
            printf("%d", coef[i]);
        else if(i == 1)
            printf("%dx", coef[i]);
        else
            printf("%dx^%d", coef[i], i);

        first = 0;
    }

    if(first) printf("0");
}

int main(void)
{
    int N, M;
    int A[101], B[101], C[201] = {0};

    scanf("%d", &N);
    for(int i = 0; i <= N; i++)
        scanf("%d", &A[i]);

    scanf("%d", &M);
    for(int i = 0; i <= M; i++)
        scanf("%d", &B[i]);

    for(int i = 0; i <= N; i++)
        for(int j = 0; j <= M; j++)
            C[i + j] += A[i] * B[j];

    printPoly(C, N + M);

    return 0;
}

골드 2. 희소 다항식 덧셈

#include <stdio.h>

typedef struct
{
    int coef;
    int exp;
} Term;

void printTerm(int coef, int exp, int *first)
{
    if(coef == 0) return;

    if(!(*first)) printf(" + ");

    if(exp == 0)
        printf("%d", coef);
    else if(exp == 1)
        printf("%dx", coef);
    else
        printf("%dx^%d", coef, exp);

    *first = 0;
}

int main(void)
{
    int N, M;
    Term A[101], B[101];

    scanf("%d", &N);
    for(int i = 0; i < N; i++)
        scanf("%d %d", &A[i].coef, &A[i].exp);

    scanf("%d", &M);
    for(int i = 0; i < M; i++)
        scanf("%d %d", &B[i].coef, &B[i].exp);

    int i = 0, j = 0;
    int first = 1;

    while(i < N && j < M)
    {
        if(A[i].exp > B[j].exp)
            printTerm(A[i].coef, A[i].exp, &first), i++;
        else if(A[i].exp < B[j].exp)
            printTerm(B[j].coef, B[j].exp, &first), j++;
        else
        {
            printTerm(A[i].coef + B[j].coef, A[i].exp, &first);
            i++;
            j++;
        }
    }

    while(i < N)
        printTerm(A[i].coef, A[i].exp, &first), i++;

    while(j < M)
        printTerm(B[j].coef, B[j].exp, &first), j++;

    if(first) printf("0");

    return 0;
}
