실버 2. 다항식 값 계산하기

다항식 P(x)의 계수와 x값이 주어질 때 P(x)의 값을 계산합니다.

P(x) = coef[0]×x⁰ + coef[1]×x¹ + ... + coef[N]×xᴺ

power 변수를 사용하면 x⁰, x¹, x², ... 을 순서대로 만들 수 있습니다.

실버 1. 다항식 덧셈

#include <stdio.h>

void printPoly(int coef[], int N)
{
    int first = 1;

    for(int i = N; i >= 0; i--)
    {
        if(coef[i] == 0) continue;

        if(!first) printf(" + ");

        if(i == 0)
            printf("%d", coef[i]);
        else if(i == 1)
            printf("%dx", coef[i]);
        else
            printf("%dx^%d", coef[i], i);

        first = 0;
    }
}

int main(void)
{
    int N;
    int A[101], B[101], C[101];

    scanf("%d", &N);

    for(int i = 0; i <= N; i++)
        scanf("%d", &A[i]);

    for(int i = 0; i <= N; i++)
        scanf("%d", &B[i]);

    for(int i = 0; i <= N; i++)
        C[i] = A[i] + B[i];

    printPoly(C, N);

    return 0;
}

실버 2. 다항식 값 계산

#include <stdio.h>

int main(void)
{
    int N;
    int coef[101];
    int x;

    scanf("%d", &N);

    for(int i = 0; i <= N; i++)
        scanf("%d", &coef[i]);

    scanf("%d", &x);

    int result = 0;
    int power = 1;

    for(int i = 0; i <= N; i++)
    {
        result += coef[i] * power;
        power *= x;
    }

    printf("%d", result);

    return 0;
}
