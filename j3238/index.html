4. EMPTY가 필요한 이유

처음에는 모든 칸이 비어 있습니다. 그런데 입력 가능한 값에는 음수도 있습니다.

값 범위: -1,000,000,000 ~ 1,000,000,000
EMPTY = -2000000001
  1. EMPTY는 실제 값보다 훨씬 작은 값입니다.
  2. 구간 최대값을 구할 때 EMPTY는 값이 없는 칸처럼 취급됩니다.
  3. 구간 전체가 EMPTY이면 출력할 값이 없으므로 아무것도 출력하지 않습니다.
  4. 삭제 명령은 그 위치를 EMPTY로 바꾸는 방식으로 구현합니다.

5. 세그먼트 트리 핵심 조건

update

idx가 구간 밖이면 return

바꾸려는 위치가 포함된 노드만 내려갑니다.

query 제외

right < start || end < left

질의 구간과 안 겹치면 EMPTY를 반환합니다.

query 사용

left <= start && end <= right

완전히 포함되면 tree[node]를 바로 사용합니다.

6. 정답 C 코드

#include <stdio.h>

#define MAXN 100000
#define EMPTY -2000000001LL

int N, M;
long long arr[MAXN + 1];
long long tree[MAXN * 4 + 4];

long long max(long long a, long long b) {
    return a > b ? a : b;
}

void init(int node, int start, int end) {
    tree[node] = EMPTY;

    if (start == end) {
        return;
    }

    int mid = (start + end) / 2;

    init(node * 2, start, mid);
    init(node * 2 + 1, mid + 1, end);
}

void update(int node, int start, int end, int idx, long long value) {
    if (idx < start || idx > end) {
        return;
    }

    if (start == end) {
        arr[idx] = value;
        tree[node] = value;
        return;
    }

    int mid = (start + end) / 2;

    update(node * 2, start, mid, idx, value);
    update(node * 2 + 1, mid + 1, end, idx, value);

    tree[node] = max(tree[node * 2], tree[node * 2 + 1]);
}

long long query(int node, int start, int end, int left, int right) {
    if (right < start || end < left) {
        return EMPTY;
    }

    if (left <= start && end <= right) {
        return tree[node];
    }

    int mid = (start + end) / 2;

    long long a = query(node * 2, start, mid, left, right);
    long long b = query(node * 2 + 1, mid + 1, end, left, right);

    return max(a, b);
}

int main() {
    int i;
    int cmd;
    int a, b;
    long long val;

    scanf("%d %d", &N, &M);

    for (i = 1; i <= N; i++) {
        arr[i] = EMPTY;
    }

    init(1, 1, N);

    for (i = 0; i < M; i++) {
        scanf("%d", &cmd);

        if (cmd == 1) {
            scanf("%d %lld", &a, &val);
            update(1, 1, N, a, val);
        }
        else if (cmd == 2) {
            scanf("%d %d", &a, &b);

            long long result = query(1, 1, N, a, b);

            if (result != EMPTY) {
                printf("%lld\n", result);
            }
        }
        else if (cmd == 3) {
            scanf("%d", &a);
            update(1, 1, N, a, EMPTY);
        }
    }

    return 0;
}
