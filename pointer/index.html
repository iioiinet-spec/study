01포인터란 무엇인가
포인터는 다른 변수의 주소를 저장하는 변수입니다. 일반 변수가 값을 저장한다면, 포인터는 그 값이 저장된 메모리 위치(주소)를 저장합니다.
C — 기본 연결
int a = 10;        // 일반 변수: 값 저장
int *p = &a;      // 포인터: a의 주소 저장

// a   → 10        (값)
// &a  → 0x100     (a의 주소)
// p   → 0x100     (a의 주소를 저장)
// *p  → 10        (p가 가리키는 실제 값)
02& 연산자 — 주소를 구한다
&변수는 그 변수가 메모리에서 저장된 위치(주소)를 반환합니다.
C — & 연산자
int a = 10;
printf("%p\n", (void*)&a);  // 예: 0x7ffd100
// &a = a가 메모리 어디에 있는지
03* 연산자 — 두 가지 의미
① 포인터 선언
int *p;
p를 포인터 변수로 선언
② 역참조 (dereference)
*p
p가 가리키는 실제 값
04역참조로 원본 값 바꾸기
*p = 값;은 포인터가 가리키는 원본 변수의 값을 직접 변경합니다.
C — 역참조 변경
int a = 10;
int *p = &a;

*p = 50;   // a = 50; 과 같은 효과

printf("%d\n", a);   // 50
printf("%d\n", *p);  // 50
출력: 50 / 50 ← 포인터로 원본 a가 바뀜
05포인터를 선언만 하면 위험!
int *p;
*p = 10; ← 위험! p가 가리키는 곳이 없음
int a = 10;
int *p = &a; ← 안전: 올바른 주소 지정 후 사용
06배열과 포인터의 관계
배열 이름은 첫 번째 원소의 주소처럼 사용됩니다. *(p+i)와 arr[i]는 같은 원소를 가리킵니다.
C — 배열·포인터 연결
int arr[3] = {10, 20, 30};
int *p = arr;      // p = &arr[0]

printf("%d\n", *p);        // 10 = arr[0]
printf("%d\n", *(p + 1)); // 20 = arr[1]
printf("%d\n", *(p + 2)); // 30 = arr[2]

예제 1 포인터로 변수 값 바꾸기
C — *p로 원본 변경
#include <stdio.h>
int main() {
    int a = 10;
    int *p = &a;

    printf("변경 전 a = %d\n", a);
    *p = 50;
    printf("변경 후 a = %d\n", a);
    printf("*p = %d\n", *p);
    return 0;
}
변경 전 a = 10
변경 후 a = 50
*p = 50
포인트: *p = 50; 은 a = 50; 과 완전히 동일한 효과입니다. p가 a의 주소를 알기 때문입니다.
예제 2 함수에서 원본 값 바꾸기
C — 포인터 매개변수
#include <stdio.h>

void changeValue(int *x) {
    *x = 100;
}

int main() {
    int num = 20;
    printf("호출 전 num = %d\n", num);
    changeValue(&num);
    printf("호출 후 num = %d\n", num);
    return 0;
}
호출 전 num = 20
호출 후 num = 100
핵심: &num으로 주소를 넘기고 → 함수 안에서 *x로 역참조해 원본 변경. call by value와의 차이를 꼭 비교하세요.
예제 3 배열과 포인터 — *(p+i) 탐색
C — 배열 포인터 탐색
#include <stdio.h>
int main() {
    int arr[3] = {10, 20, 30};
    int *p = arr;

    printf("arr[0]=%d  *p=%d\n",
           arr[0], *p);
    printf("arr[1]=%d  *(p+1)=%d\n",
           arr[1], *(p+1));
    printf("arr[2]=%d  *(p+2)=%d\n",
           arr[2], *(p+2));
    return 0;
}
arr[0]=10 *p=10
arr[1]=20 *(p+1)=20
arr[2]=30 *(p+2)=30
헷갈림 주의* 의 두 가지 얼굴
선언에서 *
int *p;   // 포인터 선언
수식에서 *
int c = a * b; // 곱셈
같은 * 기호지만 의미가 다릅니다. 선언에서는 포인터, 수식에서는 역참조 또는 곱셈으로 문맥에 따라 구분하세요.
