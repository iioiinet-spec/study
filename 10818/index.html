BOJ # 10818

최소, 최대

같은 문제를 6가지 방법으로 풀며 구조체·포인터·함수를 한 번에 학습합니다

풀이 2-1 — 기본 풀이
배열 없음 · 변수 2개
#include <stdio.h> int main(void) { int N, x; int min, max; scanf("%d", &N); scanf("%d", &x); min = x; // 첫 수로 min 초기화 max = x; // 첫 수로 max 초기화 for (int i = 1; i < N; i++) { scanf("%d", &x); if (x < min) min = x; // 더 작으면 min 갱신 if (x > max) max = x; // 더 크면 max 갱신 } printf("%d %d\n", min, max); return 0; }

핵심 포인트 기본
  • 1
    첫 수로 초기화 — 0 대신 실제 첫 번째 입력값으로 min/max를 초기화합니다. 0으로 하면 모든 수가 음수일 때 오답이 납니다.
  • 2
    배열 없이 변수 두 개(min, max)만으로 처리합니다. 입력받으면서 즉시 비교하므로 메모리가 가장 적습니다.
  • 3
    데이터를 저장하지 않아 나중에 다시 볼 수 없습니다. 최솟값·최댓값만 구하면 될 때 가장 적합합니다.
💡 왜 첫 수로 초기화? 입력 범위가 -1,000,000 ~ 1,000,000 이므로 0으로 초기화하면 모두 음수인 경우 max가 0으로 잘못 계산됩니다.
풀이 2-2 — 배열 사용
int arr[ ]
#include <stdio.h> int main(void) { int N; int arr[1000000]; // 최대 개수만큼 배열 선언 scanf("%d", &N); for (int i = 0; i < N; i++) { scanf("%d", &arr[i]); // 전체를 배열에 저장 } int min = arr[0]; // 배열 첫 원소로 초기화 int max = arr[0]; for (int i = 1; i < N; i++) { if (arr[i] < min) min = arr[i]; if (arr[i] > max) max = arr[i]; } printf("%d %d\n", min, max); return 0; }

풀이 2-3 — 함수로 분리
포인터 매개변수
#include <stdio.h> /* min, max 두 값을 동시에 반환하려면 포인터 매개변수로 주소를 받아야 한다 */ void findMinMax(int arr[], int n, int *min, int *max) { *min = arr[0]; // *min: min이 가리키는 주소에 값 씀 *max = arr[0]; for (int i = 1; i < n; i++) { if (arr[i] < *min) *min = arr[i]; if (arr[i] > *max) *max = arr[i]; } } int main(void) { int N; int arr[1000000]; int min, max; scanf("%d", &N); for (int i = 0; i < N; i++) { scanf("%d", &arr[i]); } findMinMax(arr, N, &min, &max); // &로 주소를 넘김 printf("%d %d\n", min, max); return 0; }

왜 포인터 매개변수가 필요한가 핵심

함수는 return으로 값을 하나만 돌려줄 수 있습니다. min과 max 두 개를 동시에 반환하려면 주소를 넘겨야 합니다.

❌ 값을 넘기면
findMinMax(arr, N, min, max); // 복사본 전달 // 원본은 그대로!
✅ 주소를 넘기면
findMinMax(arr, N, &min, &max); // 실제 주소 전달 // 원본 값을 직접 변경!
💡 &min은 min의 주소 / *min = ...은 그 주소에 값 쓰기. 이것이 포인터 매개변수의 핵심입니다.
풀이 2-4 — 포인터로 배열 탐색
*(p+i) 포인터 산술
#include <stdio.h> /* 배열 이름 대신 포인터로 원소에 접근하는 방법 */ void findMinMax(int *p, int n, int *min, int *max) { *min = *p; // *(p+0) == p[0] == 첫 번째 원소 *max = *p; for (int i = 1; i < n; i++) { if (*(p + i) < *min) *min = *(p + i); // *(p+i) == p[i] if (*(p + i) > *max) *max = *(p + i); } } int main(void) { int N; int arr[1000000]; int min, max; scanf("%d", &N); for (int i = 0; i < N; i++) { scanf("%d", &arr[i]); } findMinMax(arr, N, &min, &max); printf("%d %d\n", min, max); return 0; }

배열 표기 vs 포인터 표기 — 완전히 동일 동치
배열 표기 (arr[])
arr[0] → 첫 번째 원소 arr[i] → i번째 원소 &arr[i] → i번째 주소
포인터 표기 (int *p)
*p → 첫 번째 원소 *(p+i) → i번째 원소 p+i → i번째 주소
💡 arr[i]와 *(arr+i)는 완전히 동일합니다. 컴파일러는 배열 표기를 내부적으로 포인터 산술로 변환합니다.
풀이 2-5 — 구조체로 결과 반환
struct 반환
#include <stdio.h> /* 결과 두 개를 구조체 하나에 묶어 반환 */ typedef struct { int min; int max; } Result; Result findMinMax(int arr[], int n) { Result r; r.min = arr[0]; r.max = arr[0]; for (int i = 1; i < n; i++) { if (arr[i] < r.min) r.min = arr[i]; if (arr[i] > r.max) r.max = arr[i]; } return r; // 구조체 통째로 반환 } int main(void) { int N; int arr[1000000]; Result ans; scanf("%d", &N); for (int i = 0; i < N; i++) { scanf("%d", &arr[i]); } ans = findMinMax(arr, N); // 구조체 통째로 받음 printf("%d %d\n", ans.min, ans.max); return 0; }

포인터 매개변수 vs 구조체 반환 비교
풀이 2-3 (포인터 매개변수)
findMinMax(arr, N, &min, &max); // & 로 주소를 넘겨야 함 // *min = ... 로 값 쓰기
풀이 2-5 (구조체 반환)
Result r = findMinMax(arr, N); // &, * 없이 깔끔 // r.min / r.max 로 접근
💡 구조체를 반환하면 &와 * 없이 결과 두 개를 한 번에 받을 수 있어 코드가 더 읽기 쉽습니다.
풀이 2-6 — 구조체 포인터 + 동적 할당
malloc · 구조체 포인터 · ->
#include <stdio.h> #include <stdlib.h> // malloc, free /* 입력 데이터를 구조체에 저장 */ typedef struct { int value; } Data; /* 결과를 구조체에 저장 */ typedef struct { int min; int max; } Result; void findMinMax(Data *p, int n, Result *r) { r->min = p[0].value; // r->min: r이 가리키는 구조체의 min r->max = p[0].value; for (int i = 1; i < n; i++) { if ((p + i)->value < r->min) r->min = (p + i)->value; // (p+i)->value == p[i].value if ((p + i)->value > r->max) r->max = (p + i)->value; } } int main(void) { int N; Data *arr; Result ans; scanf("%d", &N); // 힙에 N개짜리 Data 배열 동적 할당 arr = (Data *)malloc(sizeof(Data) * N); for (int i = 0; i < N; i++) { scanf("%d", &arr[i].value); } findMinMax(arr, N, &ans); // &ans: Result 구조체 주소 전달 printf("%d %d\n", ans.min, ans.max); free(arr); // 반드시 해제! 안 하면 메모리 누수 return 0; }

새로 등장한 개념 4가지 심화
  • 1
    동적 할당 (malloc) — malloc(sizeof(Data) * N)으로 힙에 N개 공간을 확보합니다. 스택 한계를 넘는 대용량 배열에 필요합니다.
  • 2
    (p+i)->value — 포인터로 구조체 배열을 탐색하며 멤버에 접근합니다. p[i].value와 완전히 동일합니다.
  • 3
    Result *r — 결과 구조체도 포인터로 받아 r->min으로 원본을 직접 수정합니다.
  • 4
    free(arr) — malloc으로 확보한 메모리는 반드시 free로 반환해야 합니다. 빠뜨리면 메모리 누수가 발생합니다.
💡 모든 개념의 총집합: 구조체 + 포인터 + 동적 할당 + -> 연산자를 한 코드에서 모두 볼 수 있습니다.
