01 정수 1개 기준 오름차순
숫자
#include <stdio.h> #include <stdlib.h> /* 구조체 정의 */ typedef struct { int value; } Data; /* 비교 함수 — 오름차순 */ int cmp(const void *a, const void *b) { Data *x = (Data *)a; Data *y = (Data *)b; return x->value - y->value; } int main() { Data arr[5] = {{30}, {10}, {50}, {20}, {40}}; int n = 5; qsort(arr, n, sizeof(Data), cmp); for (int i = 0; i < n; i++) printf("%d ", arr[i].value); printf("\n"); return 0; }

value 하나짜리 구조체 x - y → 오름차순 가장 기본 패턴
비교 함수
return x->value - y->value; // 작은 값이 앞으로
📥 숫자 입력 (공백 구분)

02 x값 기준 오름차순

#include <stdio.h> #include <stdlib.h> typedef struct { int x; int y; } Point; int cmp(const void *a, const void *b) { Point *p = (Point *)a; Point *q = (Point *)b; return p->x - q->x; // x만 비교 } int main() { Point arr[5] = { {3, 9}, {1, 7}, {5, 2}, {2, 8}, {4, 1} }; int n = 5; qsort(arr, n, sizeof(Point), cmp); for (int i = 0; i < n; i++) printf("(%d, %d)\n", arr[i].x, arr[i].y); return 0; }

03 x 오름차순, x 같으면 y 오름차순
좌표 ⭐
#include <stdio.h> #include <stdlib.h> typedef struct { int x; int y; } Point; int cmp(const void *a, const void *b) { Point *p = (Point *)a; Point *q = (Point *)b; if (p->x != q->x) return p->x - q->x; // 1순위: x return p->y - q->y; // 2순위: y } int main() { Point arr[6] = { {3, 2}, {1, 5}, {3, 1}, {2, 7}, {1, 2}, {2, 4} }; int n = 6; qsort(arr, n, sizeof(Point), cmp); for (int i = 0; i < n; i++) printf("(%d, %d)\n", arr[i].x, arr[i].y); return 0; }
핵심 포인트
2단계 비교 백준 좌표 정렬 기본 if-return 패턴
비교 함수
if (p->x != q->x) return p->x - q->x; return p->y - q->y;
📥 좌표 입력 (x y, 줄바꿈)

04 x 내림차순, x 같으면 y 오름차순
좌표
#include <stdio.h> #include <stdlib.h> typedef struct { int x; int y; } Point; int cmp(const void *a, const void *b) { Point *p = (Point *)a; Point *q = (Point *)b; if (p->x != q->x) return q->x - p->x; // x 내림: q-p return p->y - q->y; // y 오름: p-q } int main() { Point arr[6] = { {3, 2}, {1, 5}, {3, 1}, {2, 7}, {1, 2}, {2, 4} }; int n = 6; qsort(arr, n, sizeof(Point), cmp); for (int i = 0; i < n; i++) printf("(%d, %d)\n", arr[i].x, arr[i].y); return 0; }
핵심 포인트
내림 → q-p 오름 → p-q
비교 함수
// 오름차순: p - q // 내림차순: q - p if (p->x != q->x) return q->x - p->x; // ↓ return p->y - q->y; // ↑
📥 좌표 입력 (x y, 줄바꿈)
📤 출력

07 길이 내림차순, 같으면 사전순
문자열 ⭐
#include <stdio.h> #include <stdlib.h> #include <string.h> typedef struct { char word[50]; int len; } Word; int cmp(const void *a, const void *b) { Word *x = (Word *)a; Word *y = (Word *)b; if (x->len != y->len) return y->len - x->len; // 길이 내림: y-x return strcmp(x->word, y->word); // 같으면 사전순 } int main() { Word arr[6] = { {"apple", 5}, {"kiwi", 4}, {"banana", 6}, {"fig", 3}, {"melon", 5}, {"grape", 5} }; int n = 6; qsort(arr, n, sizeof(Word), cmp); for (int i = 0; i < n; i++) printf("%s %d\n", arr[i].word, arr[i].len); return 0; }

길이 내림: y-x 같으면 strcmp 2단계 비교
비교 함수
if (x->len != y->len) return y->len - x->len; return strcmp(x->word, y->word);

08 국어↓ 영어↑ 수학↓ 이름↑

#include <stdio.h> #include <stdlib.h> #include <string.h> typedef struct { char name[50]; int kor, eng, math; } Student; int cmp(const void *a, const void *b) { Student *x = (Student *)a; Student *y = (Student *)b; if (x->kor != y->kor) return y->kor - x->kor; // 국어↓ if (x->eng != y->eng) return x->eng - y->eng; // 영어↑ if (x->math != y->math) return y->math - x->math; // 수학↓ return strcmp(x->name, y->name); // 이름↑ } int main() { Student arr[5] = { {"Kim", 90, 80, 70}, {"Lee", 90, 70, 90}, {"Park", 80, 80, 100}, {"Choi", 90, 70, 85}, {"Jung", 90, 70, 85} }; int n = 5; qsort(arr, n, sizeof(Student), cmp); for (int i = 0; i < n; i++) printf("%s %d %d %d\n", arr[i].name, arr[i].kor, arr[i].eng, arr[i].math); return 0; }

09 끝 시간 오름차순, 같으면 시작 시간 오름차순
회의 ⭐
#include <stdio.h> #include <stdlib.h> typedef struct { int start; int end; } Meeting; int cmp(const void *a, const void *b) { Meeting *x = (Meeting *)a; Meeting *y = (Meeting *)b; if (x->end != y->end) return x->end - y->end; // 끝 시간↑ return x->start - y->start; // 시작 시간↑ } int main() { Meeting arr[5] = { {3, 5}, {1, 4}, {2, 3}, {1, 2}, {3, 3} }; int n = 5; qsort(arr, n, sizeof(Meeting), cmp); for (int i = 0; i < n; i++) printf("%d %d\n", arr[i].start, arr[i].end); return 0; }
핵심 포인트
회의실 배정 그리디 끝 시간이 1순위 시험 빈출 패턴
비교 함수
if (x->end != y->end) return x->end - y->end; return x->start - y->start;

10 입력받아 정렬하는 실전형 예제
실전 ⭐
#include <stdio.h> #include <stdlib.h> typedef struct { int x; int y; } Point; int cmp(const void *a, const void *b) { Point *p = (Point *)a; Point *q = (Point *)b; if (p->x != q->x) return p->x - q->x; return p->y - q->y; } int main() { int n; scanf("%d", &n); /* 동적 할당 — n개를 힙에 확보 */ Point *arr = (Point *)malloc(sizeof(Point) * n); for (int i = 0; i < n; i++) scanf("%d %d", &arr[i].x, &arr[i].y); qsort(arr, n, sizeof(Point), cmp); for (int i = 0; i < n; i++) printf("%d %d\n", arr[i].x, arr[i].y); free(arr); // 반드시 해제! return 0; }
핵심 포인트
백준 스타일 malloc 동적 할당 free 필수 x↑ y↑ 정렬
